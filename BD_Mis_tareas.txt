CREATE DATABASE mistareas CHARACTER SET utf8 COLLATE utf8_spanish_ci;

CREATE TABLE Personas( ID INT NOT NULL AUTO_INCREMENT , Nombre VARCHAR(20) CHARACTER SET utf8 COLLATE utf8_spanish_ci NOT NULL , Correo VARCHAR(50) CHARACTER SET utf8 COLLATE utf8_spanish_ci NOT NULL , Clave VARCHAR(16) CHARACTER SET utf8 COLLATE utf8_spanish_ci NOT NULL ,
 PRIMARY KEY (ID)) ENGINE = InnoDB;

CREATE TABLE tareas (ID INT NOT NULL AUTO_INCREMENT, Usuario INT NOT NULL, Nombre VARCHAR(30) CHARACTER SET utf8 COLLATE utf8_spanish_ci NOT NULL, Descripcion TEXT (100) CHARACTER SET utf8 COLLATE utf8_spanish_ci NOT NULL, Estado TINYINT(1) NOT NULL DEFAULT 0, Fecha DATE NOT NULL,
PRIMARY KEY (ID)) ENGINE = InnoDB;
